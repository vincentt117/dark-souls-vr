218  template<typename T, int N>
219  inline bool CGSolvePositiveDefiniteSystem(T* x, const T* A, const T* b, T threshold) {
220 
221  ZeroVector(x, N);
222 
223  T r[N];
224  CopyVector(r, b, N);
225  T p[N];
226  CopyVector(p, r, N);
227 
228  int iter = 0;
229  while(iter < DEFAULT_CONJUGATE_GRADIENT_NUM_ITER) {
230  iter ++;
231  T alpha_num = DotProduct(r, r, N);
232  T Ap[N];
233  Multiply(Ap, A, N, N, p, N, 1);
234  T alpha_denom = DotProduct(p, Ap, N);
235 
236  ASSERT(alpha_denom != 0, "");
237 
238  T alpha = alpha_num / alpha_denom;
239  T r_sum = 0;
240  T beta_old = DotProduct(r, r, N);
241  ASSERT(beta_old > 0, "");
242 
243  for(int i=0;i<N;i++) {
244  x[i] = x[i] + alpha * p[i];
245  r[i] = r[i] - alpha * Ap[i];
246  r_sum += r[i]*r[i];
247  }
248 
249  if(r_sum < threshold) {
250  break;
251  }
252 
253  T beta_new = DotProduct(r, r, N);
254  T beta = beta_new / beta_old;
255 
256  for(int i=0;i<N;i++) {
257  p[i] = r[i] + beta * p[i];
258  }
259 
260  }
261 
262  return true;
263  }

340  bool Cholesky_Blocked(T* L, T* A, int n, int b, T threshold = 0) {
341  //ASSERT(n%b == 0, "Blocks must fit evenly into n");
342  int num_iters = n/b;
343  for (int i = 0; i < num_iters; i++) {
344  // L_ii = Cholesky(A_ii)
345 
346  // initialize top left blocks
347  T L_ii[b*b];
348  T A_ii[b*b];
349  // load top left block of A into A_ii
350  LoadSquareBlockFromMatrix(&A_ii[0], &A[0], n, n, b, i, i);
351  // take cholesky of A_ii and place it in L_ii
352  Cholesky(L_ii, b, A_ii, b, b);
353  // Write L_ii into L, still save L_ii for use
354  LoadBlockToMatrix(&L_ii[0], &L[0], n, n, b, i, i);
355 
356  // L_bl = A_bl * L_tl^T
357  T L_iiINVT[b*b];
358  // invert the top left block
359  InvertLeftTriangularMatrixTranspose(L_iiINVT, &L_ii[0], b);
360  for (int j=i+1; j < num_iters; j++) {
361  //T* L_ji = &L[n*j*b + b*i];
362  T L_ji[b*b];
363 
364  // Copy A_ji block to correct blocksize
365  T A_ji[b*b];
366  LoadSquareBlockFromMatrix(&A_ji[0], &A[0], n, n, b, j, i);
367 
368  // Possibly optimize by having this write to the big matrix when it computes
369  Multiply(L_ji, A_ji, b, b, L_iiINVT, b, b);
370  LoadBlockToMatrix(&L_ji[0], &L[0], n, n, b, j, i);
371  }
372  // A_jk -= A_ji*A_ki^T
373  for (int j = i+1; j < num_iters; j++) {
374  // allocate and loadanchor block
375  T L_ji[b*b];
376  LoadSquareBlockFromMatrix(&L_ji[0], &L[0], n, n, b, j, i);
377  for (int k = i+1; k < num_iters; k++) {
378  // allocate outerproduct block and result block
379  T A_jk[b*b];
380  T L_ki[b*b];
381  // load outerproduct block and result block
382  LoadSquareBlockFromMatrix(&A_jk[0], &A[0], n, n, b, j, k);
383  LoadSquareBlockFromMatrix(&L_ki[0], &L[0], n, n, b, k, i);
