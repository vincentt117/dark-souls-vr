46 // ASSERT(yp >= 0 && yp < height, "yp out of bounds");

51  // Pointer to 2 image rows

53  const Tin* p1 = (const Tin*)((const unsigned char*)p0+step);
54 
55  // Compute weights
56  const Tout w0 = (xp_plus_1-x) * (yp_plus_1-y);
57  const Tout w1 = (x-xp) * (yp_plus_1-y);
58  const Tout w2 = (xp_plus_1-x) * (y-yp);
59  const Tout w3 = (x-xp) * (y-yp);

61 // ASSERT(w0 >= 0 && w0 <= 1.0001, "Out of range");
62 // ASSERT(w1 >= 0 && w1 <= 1.0001, "Out of range");
63 // ASSERT(w2 >= 0 && w2 <= 1.0001, "Out of range");
