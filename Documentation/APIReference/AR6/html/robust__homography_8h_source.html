117  int max_trials = HOMOGRAPHY_DEFAULT_MAX_TRIALS,
118  int chunk_size = HOMOGRAPHY_DEFAULT_CHUNK_SIZE) {

120  int* point_perm;
121  T one_over_scale2;
122  T min_cost;
123  int num_hypotheses, num_hypotheses_remaining, min_index;
124  int cur_chunk_size, next_chunk, this_chunk_end;
125  int trial;
126  int seed;
127  int sample_size = 4;
128 
129 // ASSERT(hyp.size() >= 9*max_num_hypotheses, "hyp vector should be of size 9*max_num_hypotheses");
130 // ASSERT(tmp_i.size() >= 2*num_points, "tmp_i vector should be of size 2*num_points");
131 // ASSERT(hyp_costs.size() >= max_num_hypotheses, "hyp_costs vector should be of size max_num_hypotheses");
132 
133  // We need at least SAMPLE_SIZE points to sample from
134  if(num_points < sample_size) {
135  return false;
136  }
137 
138  seed = 1234;
139 
140  hyp_perm = &tmp_i[0];
141  point_perm = &tmp_i[num_points];
142 
143  one_over_scale2 = 1/sqr(scale);
144  chunk_size = min2(chunk_size, num_points);
145  next_chunk = 0;
146

148  SequentialVector(hyp_perm, num_points, 0);
149  SequentialVector(point_perm, num_points, 0);
150 
151  // Shuffle the indices
152  ArrayShuffle(hyp_perm, num_points, num_points, seed);
153  ArrayShuffle(point_perm, num_points, num_points, seed);
154 
155  // Compute a set of hypotheses
156  for(trial = 0, num_hypotheses = 0;
157  trial < max_trials && num_hypotheses < max_num_hypotheses;
158  trial++) {
159 
160  // Shuffle the first SAMPLE_SIZE indices
161  ArrayShuffle(hyp_perm, num_points, sample_size, seed);
162 
163  // Check if the four points are geometrically valid
164  if(!Homography4PointsGeometricallyConsistent(&p[hyp_perm[0]<<1],
165  &p[hyp_perm[1]<<1],
166  &p[hyp_perm[2]<<1],
167  &p[hyp_perm[3]<<1],
168  &q[hyp_perm[0]<<1],
169  &q[hyp_perm[1]<<1],
170  &q[hyp_perm[2]<<1],
171  &q[hyp_perm[3]<<1])) {
172  continue;
173  }
174 
175  // Compute the homography
176  if(!SolveHomography4Points(&hyp[num_hypotheses*9],
177  &p[hyp_perm[0]<<1],
178  &p[hyp_perm[1]<<1],
179  &p[hyp_perm[2]<<1],
180  &p[hyp_perm[3]<<1],
181  &q[hyp_perm[0]<<1],
182  &q[hyp_perm[1]<<1],
183  &q[hyp_perm[2]<<1],
184  &q[hyp_perm[3]<<1])) {
185  continue;
186  }
187 
188  // Check the test points
189  if(num_test_points > 0) {
190  if(!HomographyPointsGeometricallyConsistent(&hyp[num_hypotheses*9], test_points, num_test_points)) {
191  continue;
192  }
193  }
194 
195  num_hypotheses++;
196  }
197 
198  // We fail if no hypotheses could be computed
199  if(num_hypotheses == 0) {
200  return false;
201  }
202 
203  // Initialize the hypotheses costs
204  for(int i = 0; i < num_hypotheses; i++) {
205  hyp_costs[i].first = 0;
206  hyp_costs[i].second = i;
207  }
208 
209  num_hypotheses_remaining = num_hypotheses;
210  cur_chunk_size = chunk_size;
211 
212  for(int i = 0;
213  i < num_points && num_hypotheses_remaining > 2;
214  i+=cur_chunk_size) {
215 
216  // Size of the current chunk
217  cur_chunk_size = min2(chunk_size, num_points-i);
218 
219  // End of the current chunk
220  this_chunk_end = i+cur_chunk_size;
221 
222  // Score each of the remaining hypotheses
223  for(int j = 0; j < num_hypotheses_remaining; j++) {
224  const T* H_cur = &hyp[hyp_costs[j].second*9];
225  for(int k = i; k < this_chunk_end; k++) {
226  hyp_costs[j].first += CauchyProjectiveReprojectionCost(H_cur,
227  &p[point_perm[k]<<1],
228  &q[point_perm[k]<<1],
229  one_over_scale2);
230  }
231  }
232 
233  // Cut out half of the hypotheses
234  FastMedian(&hyp_costs[0], num_hypotheses_remaining);
235  num_hypotheses_remaining = num_hypotheses_remaining>>1;
236  }
237 
238  // Find the best hypothesis
239  min_index = hyp_costs[0].second;
240  min_cost = hyp_costs[0].first;
241  for(int i = 1; i < num_hypotheses_remaining; i++) {
242  if(hyp_costs[i].first < min_cost ) {
243  min_cost = hyp_costs[i].first;
244  min_index = hyp_costs[i].second;
245  }
246  }
247 
248  // Move the best hypothesis
249  CopyVector9(H, &hyp[min_index*9]);
250  NormalizeHomography(H);
251 
252  return true;
253  }

301  inline void HomographyJacobianDirect(T J[16], T f[2], const T pp[2], const T p[2], const T q[2], const T H[9]) {
302  //ASSERT(std::abs(1-H[8]) < 1e-5, "H8 must be 1");
303 
304  // Linearization point
305  const T& x = p[0];
306  const T& y = p[1];
307 
308  T w = H[6]*x + H[7]*y + H[8];
309  T one_over_w = SafeReciprocal(w);
310  T one_over_w_sqr = sqr(one_over_w);
311 
312  T xx = H[0]*x + H[1]*y + H[2];
313  T yy = H[3]*x + H[4]*y + H[5];
314 
315  J[0] = x*one_over_w;
316  J[1] = y*one_over_w;
317  J[2] = one_over_w;
318  J[3] = 0;
319  J[4] = 0;
320  J[5] = 0;
321  J[6] = -x*xx*one_over_w_sqr;
322  J[7] = -y*xx*one_over_w_sqr;
323  J[8] = 0;
324  J[9] = 0;
325  J[10] = 0;
326  J[11] = J[0];
327  J[12] = J[1];
328  J[13] = J[2];
329  J[14] = -x*yy*one_over_w_sqr;
330  J[15] = -y*yy*one_over_w_sqr;
331 
332  f[0] = pp[0]-q[0];
333  f[1] = pp[1]-q[1];
334  }

422  bool PolishHomography(T H[9],
423  const T* p,
424  const T* q,
425  int num_points,
426  T scale = HOMOGRAPHY_DEFAULT_CAUCHY_SCALE,
427  int max_iterations = HOMOGRAPHY_DEFAULT_MAX_NUM_ITERATIONS,
428  int max_stops = HOMOGRAPHY_DEFAULT_MAX_STOPS,
429  T improvement = HOMOGRAPHY_DEFAULT_IMPROVEMENT_TOL)
430  {
431  T one_over_scale2;
432  T JtJ[64];
433  T reg_JtJ[64];
434  T Jtr[8];
435  T delta[8];
436  T cost, last_cost;
437  bool update;
438  T lambda;
439  T Hp[9];
440  int stops;
441  int num_params;
442 
443  stops = 0;
444  lambda = 0.01;
445  update = true;
446  one_over_scale2 = 1/(scale*scale);
447 
448  num_params = 8;
449 
450  NormalizeHomography(H);
451 
452  last_cost = CauchyProjectiveReprojectionCost(H, p, q, num_points, one_over_scale2);
453 
454  for(int i = 0; i < max_iterations && stops < max_stops; i++) {
455  if(update) {
456  ComputeDirectHomographyNormalEquationsPostMultiply(JtJ, Jtr, H, p, q, num_points, one_over_scale2);
457  CopyVector(reg_JtJ, JtJ, num_params*num_params);
458  }
459

462  //

465  return false;
466  }
467  UpdateDirectProjectiveMotionPostMultiply(Hp, H, delta);
468 
469  //
470  // Compute robust error function
471  //
472  cost = CauchyProjectiveReprojectionCost(Hp, p, q, num_points, one_over_scale2);
473 
474  if(cost < last_cost) {
475  CopyVector9(H, Hp);
476  stops = (last_cost - cost) < improvement ? stops + 1 : 0;
477  last_cost = cost;
478  lambda = max2<T>(lambda*0.1, 0.000001);
479  update = true;
480  }
481  else {
482  lambda = min2<T>(lambda*10, 100000);
483  stops++;
484  update = false;
485  }
486  }
487 
488  return true;
489  }
