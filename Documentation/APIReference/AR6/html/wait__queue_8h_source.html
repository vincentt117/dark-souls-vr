174  template<typename T>
175  bool WaitQueue<T>::push(T const& item) {
176  bool ret = false;
177 
178  Lockit lock(mPushMutex);
179 
180  // Check size constraint on the queue
181  if(mQueue.size() >= mMaxSize) {
182  switch(mOverflowAction) {
183  case kOverflowActionDrop:
184  break;
185  case kOverflowActionBlock:
186  do {
187  // Here we want to wait until another thread pops something
188  // from the queue. We must lock before checking the size of the
189  // queue.
190  mPopGuard.lock();
191  if(mQueue.size() < mMaxSize) {
192  mPopGuard.unlock();
193  break;
194  }
195  mPopGuard.wait(0.001);
196  mPopGuard.unlock();
197  } while(!mCancel);
198  mCancel = false;
199  break;
200  case kOverflowActionPop:
201  // Pop items from the queue to make room for one push.
202  mPopMutex.lock();
203  while(mQueue.size() >= mMaxSize) {
204  mQueue.drop();
205  }
206  mPopMutex.unlock();
207  }
208  }
209 
210  if(mQueue.size() < mMaxSize) {
211  ret = mQueue.push(item);
212  //ASSERT(ret, "Failed to push item onto queue");
213  if(ret) {
214  mPushGuard.lock();
215  mPushGuard.signal();
216  mPushGuard.unlock();
217  }
218  }
219 
220  return ret;
221  }

223  template<typename T>
224  bool WaitQueue<T>::pop(T& item) {
225  bool ret;
226 
227  Lockit lock(mPopMutex);
228 
229  // Try to pop an item from the queue
230  ret = mQueue.pop(item);
231  if(ret == false) {
232  // The queue is empty, so we need to lock the push guard to ensure we are signaled
233  // when an item is pushed.
234  mPushGuard.lock();
235  // Re-check after the guard is locked if queue is empty before "waiting"
236  if(mQueue.size() == 0){
237  mPushGuard.wait(mPopWaitTimeout);
238  }
239  ret = mQueue.pop(item);
240  mPushGuard.unlock();
241  }
242 
243  // If we popped an item, then we need to signal waiting threads on push(). We
244  // only need to do this if the overflow action is blocking.
245  if(ret && mOverflowAction == kOverflowActionBlock) {
246  mPopGuard.lock();
247  mPopGuard.signal();
248  mPopGuard.unlock();
249  }
250 
251  return ret;
252  }
