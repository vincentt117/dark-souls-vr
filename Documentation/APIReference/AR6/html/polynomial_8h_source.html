156  inline void PolyDiv(std::deque<T>& Q, std::deque<T>& R, const std::deque<T>& A, const std::deque<T>& B) {
157  ASSERT_VALID_POLY(A);
158  ASSERT_VALID_POLY(B);
159  ASSERT(!(B.size() == 1 && B[0] == 0.0), "cannot divide by zero");
160  R.clear();
161  Q.clear();
162  if (B.size() > A.size()) {
163  Q.push_back(0.0);
164  R = A;
165  }
166  else {
167  if (B.size() == 1) {
168  PolyScale(Q, A, B[0]);
169  R.push_back(0);
170  }
171  // generic case
172  else {
173  int index = A.size() - B.size();
174  ASSERT(R.size() == 0, "should be empty at this point");
175  // R = A[len(A) - len(B):]
176  for (int i = A.size() - B.size(); i < A.size(); i++) {
177  R.push_back(A[i]);
178  }
179  std::deque<T> to_sub;
180  std::deque<T> sub_res;
181  for (int K = 0; K < A.size() - B.size() + 1; K++) {
182  T multiple = (R.back() != 0) ? R.back()/B.back() : 0;
183  Q.push_front(multiple);
184  PolyScale(to_sub, B, multiple);
185 
186  PolySub(sub_res, R, to_sub);
187  // if the result did not have exactly zero in its last component
188  if (sub_res.size() == R.size()) {
189  // WZ/April 28: temporarily commented out
190 // ASSERT(std::abs(sub_res.back()/sub_res.front()) < 1e-6, "this must only happen in case of numerical roundoff error");
191  sub_res.pop_back();
192  }
193  const int num_zeros_to_add = R.size() - sub_res.size() - 1;
194  K = K + num_zeros_to_add;
195  // if it was decreased to zero then add one more to K
196  if (sub_res.size() == 1 && sub_res[0] == 0.0) {
197  K++;
198  }
199  for (int i = 0; i < num_zeros_to_add; i++) {
200  Q.push_front(0.0);
201  }
202  R = sub_res;
203  ASSERT_VALID_POLY(R);
204  index--;
205  if (index > -1) {
206  R.push_front(A[index]);
207  }
208  // make it valid
209  CutLeadingZeros(R);
210  }
211  }

318  inline void FindZeros(std::vector<SearchInterval<T>>& intervals,
319  const std::vector<std::deque<T>>& sturm_chain,
320  const T search_start, const T search_end,
321  const T tol) {
322  const int max_zeros = sturm_chain.size();
323  std::vector<T> evaluated_chain;
324  intervals.resize(max_zeros);
325 
326  // fill in the first search interval
327  SearchInterval<T>& first_interval = intervals[0];
328  first_interval.interval_start = search_start;
329  first_interval.interval_end = search_end;
330  // evaluate the chain at the first interal and fill in sign changes
331  EvaluateChain(evaluated_chain, sturm_chain, search_start);
332  first_interval.interval_start_sign_changes = SignChanges(evaluated_chain);
333  // evaluate the chain at the end of the first interval and fill in sign changes
334  EvaluateChain(evaluated_chain, sturm_chain, search_end);
335  first_interval.interval_end_sign_changes = SignChanges(evaluated_chain);
336  // compute number of zeros for first interval
337  first_interval.num_zeros = first_interval.interval_start_sign_changes - first_interval.interval_end_sign_changes;
338 
339  // the number of intervals that we are currently searching through
340  int num_intervals = 1;
341  bool go = true;
342  while (go) {
343  // the number of intervals at the beginning of this iteration
344  int num_intervals_at_this_iteration = num_intervals;
345  for (int i = 0; i < num_intervals_at_this_iteration; i++) {
346 
347  // get the interval
348  SearchInterval<T>& cur_interval = intervals[i];
349  T a = cur_interval.interval_start;
350  T b = cur_interval.interval_end;
351  // if the interval is big enough, split it in half
352  if ((b - a) > tol) {
353  T c = (a+b)/2;
354 
355  ASSERT(cur_interval.interval_start_sign_changes != -1, "sign changes should have been computed already");
356  ASSERT(cur_interval.interval_end_sign_changes != -1, "sign changes should have been computed already");
357 
358  // evaluate the chain at c
359  EvaluateChain(evaluated_chain, sturm_chain, c);
360  bool try_again = false;
361  for (int v = 0; v < evaluated_chain.size(); v++) {
362  if (evaluated_chain[v] == 0) {
363  try_again = true;
364  }
365  }
366  // move c and re-evaluate the chain if it equals 0 anywhere
367  while (try_again) {
368  try_again = false;
369  // move c slightly
370  c = c + tol;
371  EvaluateChain(evaluated_chain, sturm_chain, c);
372  for (int v = 0; v < evaluated_chain.size(); v++) {
373  if (evaluated_chain[v] == 0) {
374  try_again = true;
375  }
376  }
377  }
378  // get sign changes at c
379  const int sign_changes_c = SignChanges(evaluated_chain);
380  // get zero counts for each sub interval
381  const int interval_1_cnt = cur_interval.interval_start_sign_changes - sign_changes_c;
382  const int interval_2_cnt = sign_changes_c - cur_interval.interval_end_sign_changes;
383 
384  // if there were no zeros in the second interval
385  if (interval_1_cnt == cur_interval.num_zeros) {
386  // move the end of this interval to c and do not create another interval
387  cur_interval.interval_end = c;
388  cur_interval.interval_end_sign_changes = sign_changes_c;
389  }
390  // if there were no zeros in teh first interval
391  else if (interval_2_cnt == cur_interval.num_zeros) {
392  // move the beginning of this interval to c and do not create another interval
393  cur_interval.interval_start = c;
394  cur_interval.interval_start_sign_changes = sign_changes_c;
395  }
396  // if we split the zeros between the intervals
397  // condense the lower interval into this one and create a new interval
398  // adding it to the back of the array of intervals
399  else {
400  const int sign_changes_b = cur_interval.interval_end_sign_changes;
401 
402  cur_interval.interval_end = c;
403  cur_interval.num_zeros = interval_1_cnt;
404  cur_interval.interval_end_sign_changes = sign_changes_c;
405  // make new interval
406  num_intervals++;
407  SearchInterval<T>& new_interval = intervals[num_intervals-1];
408  new_interval.interval_start = c;
409  new_interval.interval_end = b;
410  new_interval.num_zeros = interval_2_cnt;
411  new_interval.interval_start_sign_changes = sign_changes_c;
412  new_interval.interval_end_sign_changes = sign_changes_b;
413  }
414  }
415  }
416  bool finished = true;
417  for (int i = 0; i < num_intervals; i++) {
418  SearchInterval<T>& interval = intervals[i];
419  if (interval.interval_end - interval.interval_start > tol) {
420  finished = false;
421  }
422  }
423  go = !finished;
424  }
425  intervals.resize(num_intervals);
426  }

464  inline bool Quadratic3Points(T& A,
465  T& B,
466  T& C,
467  const T p1[2],
468  const T p2[2],
469  const T p3[2]) {
470  T d1 = (p3[0]-p2[0])*(p3[0]-p1[0]);
471  T d2 = (p1[0]-p2[0])*(p3[0]-p1[0]);
472  T d3 = p1[0]-p2[0];
473 
474  // If any of the denominators are zero then return FALSE.
475  if(d1 == 0 ||

477  d3 == 0) {
478  A = 0;
479  B = 0;
480  C = 0;
481  return false;

485  T b = p2[0]*p2[0];

488  A = ((p3[1]-p2[1])/d1)-((p1[1]-p2[1])/d2);
489  B = ((p1[1]-p2[1])+(A*(b-a)))/d3;
490  C = p1[1]-(A*a)-(B*p1[0]);
491  return true;
492  }
