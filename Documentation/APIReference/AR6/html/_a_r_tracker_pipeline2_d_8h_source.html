113  unsigned char* drawFeatures(std::shared_ptr<unsigned char*> referenceImage, int imgWidth, int imgHeight, int markerId, std::string fileName, std::string saveName)
114  {
115  size_t width = imgWidth;
116  size_t height = imgHeight;
117 
118  artvision::ImageDynamicUINT8 targetImage = artvision::ImageDynamicUtils::view<artvision::ImageType::UINT8>(height, width, width, *referenceImage.get(), width*height);
119  std::shared_ptr<artvision::ImageDynamicUINT8> targetImagePtr = std::make_shared<artvision::ImageDynamicUINT8>(targetImage);
120  artvision::Target target(targetImagePtr, markerId);
121 
122  artvision::Image bgrOutputImage;
123  artvision::Image refImageOldGray = artvision::convertToGrayScaleOldImage(targetImage);
124  artvision::Gray2BGR(bgrOutputImage, refImageOldGray);
125 
126  //Create identity matrix for homography
127  using homography_t = artvision::Homography<float>;
128  homography_t homography;
129  homography.row(0)[0] = 1;
130  homography.row(0)[1] = 0;
131  homography.row(0)[2] = 0;
132  homography.row(1)[0] = 0;
133  homography.row(1)[1] = 1;
134  homography.row(1)[2] = 0;
135  homography.row(2)[0] = 0;
136  homography.row(2)[1] = 0;
137  homography.row(2)[2] = 1;
138 
139  artvision::RecognisedPlanarTarget recotarget(homography, markerId);
140 
141  artvision::TrackedPlanarTarget::projection_matrix_t projMatrix;
142  artvision::TrackedPlanarTarget trackedTarget(projMatrix, homography, markerId);
143 
144  //Create new tracker instance to avoid interrupting any current tracking threads.
145  artvision::SingleTargetPlanarTracker tracker(std::vector<artvision::Target>({ target }), m_2DTrackerLensModel, height, width);
146  tracker.initializeFromSingleTargetPlanarRecogniser(recotarget);
147  tracker.drawTemplates(trackedTarget, bgrOutputImage);
148 
149  m_2DRecogniser->drawFeatures(recotarget, bgrOutputImage);
150  return bgrOutputImage.get();
151  }
