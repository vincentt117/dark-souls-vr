103  inline void ConstructFischlerBollesPolynomialOriginal(T D[5],
104  const T a2, const T b2, const T c2,
105  const T cos_alpha, const T cos_beta, const T cos_gamma) {
106  T cos_alpha2 = sqr(cos_alpha);
107  T cos_beta2 = sqr(cos_beta);
108 
109  T K1 = a2/b2;
110  T K2 = a2/c2;
111 
112  T KK = K1*K2;
113  T KKm = KK-K1-K2;
114  T KKp1 = KK+K1-K2;
115  T KKp2 = KK+K2-K1;
116 
117  D[0] = sqr(KKm) - 4*KK*cos_alpha2;
118  D[1] = 4*KKm*K2*(1-K1)*cos_gamma + 4*K1*cos_alpha*(KKp2*cos_beta
119  + 2*K2*cos_gamma*cos_alpha);
120  D[2] = sqr(2*K2*(1-K1)*cos_gamma) + 2*KKp1*KKm + 4*K1*((K1-K2)*cos_alpha2+(1-K2)*K1*cos_beta2
121  -2*K2*(1+K1)*cos_gamma*cos_beta*cos_alpha);
122  D[3] = 4*KKp1*K2*(1-K1)*cos_gamma + 4*K1*(KKp2*cos_beta*cos_alpha + 2*K1*K2*cos_gamma*cos_beta2);
123  D[4] = sqr(KKp1) - 4*sqr(K1)*K2*cos_beta2;
124  }

128  inline void ConstructFischlerBollesPolynomial(T D[5],
129  const T a2, const T b2, const T c2,
130  const T cos_alpha, const T cos_beta, const T cos_gamma) {
131  T cos_alpha2 = sqr(cos_alpha);
132  T cos_beta2 = sqr(cos_beta);
133  T cos_gamma2 = sqr(cos_gamma);
134 
135  T b2mc2 = b2-c2;
136  T a2mb2 = a2-b2;
137  T a2mb2mc2 = a2-b2-c2;
138  T a2mb2pc2 = a2-b2+c2;
139 
140  D[0] = 4*b2*c2*cos_alpha2-sqr(a2mb2mc2);
141  D[1] = -4*c2*(a2+b2mc2)*cos_alpha*cos_beta
142  -8*b2*c2*cos_alpha2*cos_gamma
143  +4*a2mb2mc2*a2mb2*cos_gamma;
144  D[2] = 4*c2*(a2-c2)*cos_beta2
145  +8*c2*(a2+b2)*cos_alpha*cos_beta*cos_gamma
146  +4*c2*b2mc2*cos_alpha2
147  -2*a2mb2mc2*a2mb2pc2
148  -4*sqr(a2mb2)*cos_gamma2;
149  D[3] = -8*a2*c2*cos_beta2*cos_gamma
150  -4*c2*b2mc2*cos_alpha*cos_beta
151  -4*a2*c2*cos_alpha*cos_beta
152  +4*a2mb2*a2mb2pc2*cos_gamma;
153  D[4] = 4*a2*c2*cos_beta2-sqr(a2mb2pc2);
154  }

176  inline int PolynomialRootsToLegs(T la[4], T lb[4], T lc[4], T y[4],
177  T roots[4],
178  const int num_roots,
179  const T a2, const T b2, const T c2,
180  const T cos_alpha, const T cos_beta, const T cos_gamma, const T threshold = 1e-12) {
181  int unique_num_roots = 0;
182 
183  unique_num_roots = UniqueVector(roots, num_roots, threshold);
184 
185  T a, b, c;
186  a = sqrt(a2);
187  b = sqrt(b2);
188  c = sqrt(c2);
189 
190  int j = 0;
191  for (int i = 0; i < unique_num_roots; i++) {
192 
193  T tmp_la, tmp_lb;
194 
195  T x = roots[i];
196 
197  tmp_la = c/std::sqrt(sqr(x) - 2*x*cos_gamma + 1);
198  tmp_lb = tmp_la*x;
199 
200  T K1 = a2/b2;
201  T K2 = a2/c2;
202 
203  T m = 1 - K1;
204  T q = sqr(x) - K1;
205  T p = 2*(K1*cos_beta - x*cos_alpha);
206 
207  T mp = 1;
208  T qp = sqr(x)*(1-K2) + 2*x*K2*cos_gamma - K2;
209  T pp = -2*x*cos_alpha;
210  if(std::abs(mp*q - m*qp) > threshold) {
211 
212  y[j] = (pp*q - p*qp)/(m*qp - mp*q);
213 
214  if(y[j]>0) {
215 
216  lc[j] = y[j]*tmp_la;
217  la[j] = tmp_la;
218  lb[j] = tmp_lb;
219 
220  j++;
221  }
222  } else {
223 
224  T y_roots[2];
225  int y_num_roots;
226  T error[2];
227 
228  y_num_roots = SolveQuadraticPolynomial(y_roots, (T)1, -2*cos_beta, (T)1-b2/sqr(tmp_la)); // A5
229 
230  if(y_num_roots>0) {
231 
232  lc[j] = y_roots[0]*tmp_la;
233  error[0] = std::abs(a2 - sqr(tmp_lb) - sqr(lc[j]) +
234  2*tmp_lb*lc[j]*cos_alpha); // A3
235 
236  if((error[0] < threshold) && (y_roots[0] > 0)){
237 
238  y[j] = y_roots[0];
239 
240  la[j] = tmp_la;
241  lb[j] = tmp_lb;
242 
243  j++;
244 
245  }
246 
247  if(y_num_roots > 1) {
248 
249  lc[j] = y_roots[1]*tmp_la;
250  error[1] = std::abs(a2 - sqr(tmp_lb) - sqr(lc[j]) +
251  2*tmp_lb*lc[j]*cos_alpha); // A3
252 
253  if ((error[1] < threshold) && (y_roots[1] > 0)) {
254 
255  y[j] = y_roots[1];
256 
257  la[j] = tmp_la;
258  lb[j] = tmp_lb;
259 
260  j++;
261  }
262 
263  }
264  } // y_num_roots > 0
265  }
266  } // i
267 
268  return j;
269  }
