219  inline void locate(int& octave,
220  int& scale,
221  float sigma) const {
222  // octave = floor(log2(s))
223  octave = (int)std::floor(artvision_instanton::log2<float>(sigma));
224  // scale = logk(s/2^octave)
225  // Here we ROUND the scale to an integer value
226  float fscale = std::log(sigma/(float)(1<<octave))*mOneOverLogK;
227  scale = (int)artvision_instanton::round(fscale);
228 
229  // The last scale in an octave has the same sigma as the first scale
230  // of the next octave. We prefer coarser octaves for efficiency.
231  if(scale == mNumScalesPerOctave-1) {
232  // If this octave is out of range, then it will be clipped to
233  // be in range on the next step below.
234  octave = octave+1;
235  scale = 0;
236  }
237 
238  // Clip the octave/scale to be in range of the pyramid
239  if(octave < 0) {
240  octave = 0;
241  scale = 0;
242  } else if(octave >= mNumOctaves) {
243  octave = mNumOctaves-1;
244  scale = mNumScalesPerOctave-1;
245  }
246 //
247 // ASSERT(octave >= 0, "Octave must be positive");
248 // ASSERT(octave < mNumOctaves, "Octave must be less than number of octaves");
249 // ASSERT(scale >= 0, "Scale must be positive");
250 // ASSERT(scale < mNumScalesPerOctave, "Scale must be less than number of scale per octave");
251  }
