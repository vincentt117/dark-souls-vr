75  bool CubicSplineSolver<T>::solve(const T* t, const T* x, int size) {
76  if(size < 3) {
77  return false;
78  }
79 
80  // Allocation
81  m_a .resize(size-1, 1);
82  m_b .resize(size);
83  m_c .resize(size, 1); // over allocated by 1
84  m_v .resize(size);
85  mSplines .resize(size-1);
86 
87  // Main diagonal vector: [2,4,4,...,4,2]
88  m_b[0] = m_b[size-1] = 2;
89  for(size_t i = 1; i < m_b.size()-1; i++) {
90  m_b[i] = 4;
91  }
92 
93  // RHS
94  m_v[0] = 3*(x[1]-x[0]);
95  m_v[size-1] = 3*(x[size-1]-x[size-2]);
96  for(int i = 1; i < size-1; i++) {
97  m_v[i] = 3*(x[i+1] - x[i-1]);
98  }
99 
100  // Solve the system
101  if(!SolveTridiagonalDestructive(&m_v[0], size, &m_a[0], &m_b[0], &m_c[0])) {
102  ARLOGe("SolveTridiagonalDestructive failed");
103  return false;
104  }
105 
106  // Form the splines
107  for(size_t i = 0; i < m_v.size()-1; i++) {
108  mSplines[i].t0 = t[i];
109  mSplines[i].t1 = t[i+1];
110  mSplines[i].a = x[i];
111  mSplines[i].b = m_v[i];
112  mSplines[i].c = 3*(x[i+1]-x[i])-2*m_v[i]-m_v[i+1];
113  mSplines[i].d = 2*(x[i]-x[i+1])+m_v[i]+m_v[i+1];
114  }
115 
116  return true;
117  }

150  template<typename T>
151  inline T CubicSplineSolver<T>::evalulate(const CubicSpline<T>& s, T t) const {
152  ASSERT(t >= s.t0, "t is out of range");
153  ASSERT(t <= s.t1, "t is out of range");
154  ASSERT(s.t1 > s.t0, "something strange with t0 and t1");
155  T dt = (t-s.t0)/(s.t1-s.t0);
156  return s.a+(s.b*dt)+(s.c*dt*dt)+(s.d*dt*dt*dt);
157  }
