26 inline void UndistortPointOulu(T x[2], const T xd[2], const T lens_coeffs[5], const int num_iterations) {
27  const T& k1 = lens_coeffs[0];
28  const T& k2 = lens_coeffs[1];
29  const T& k3 = lens_coeffs[4];
30  const T& p1 = lens_coeffs[2];
31  const T& p2 = lens_coeffs[3];
32 
33  artvision_instanton::CopyVector2(x, xd);
34  for(int i = 0; i < num_iterations; i++) {
35  T r_2 = artvision_instanton::DotProduct2(x, x);
36  T k_radial = 1 + k1*r_2 + k2*r_2*r_2 + k3*r_2*r_2*r_2;
37  T delta_x0 = 2*p1*x[0]*x[1] + p2*(r_2 + 2*x[0]*x[0]);
38  T delta_x1 = p1 * (r_2 + 2*x[1]*x[1]) + 2*p2*x[0]*x[1];
39  x[0] = artvision_instanton::SafeDivision(xd[0]-delta_x0, k_radial);
40  x[1] = artvision_instanton::SafeDivision(xd[1]-delta_x1, k_radial);
41  }
42 }
