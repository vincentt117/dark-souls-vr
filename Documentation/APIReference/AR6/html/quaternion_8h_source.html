58  template<typename T>
59  inline void QuaternionSLERP(T q[4], const T q0[4], const T q1[4], T t) {
60  T qq1[4];
61 
62  // "t" is the range [0,1]
63  //ASSERT(t >= 0, "Out of range");
64  //ASSERT(t <= 1, "Out of range");
65 
66  // Take the shortest path: (q and -q represent the same rotation)
67  T d = DotProduct4(q0, q1);
68  if(d < 0) {
69  ScaleVector4<T>(qq1, q1, -1);
70  d *= -1;
71  } else {
72  CopyVector4(qq1, q1);
73  }
74 
75  // Make sure that "d" is in the range [-1,1].
76  //ASSERT(d <= 1+1e-5, "Out of range");
77  //ASSERT(d >= -1-1e-5, "Out of range");
78  d = ClipScalar<T>(d, -1, 1);
79 
80  // Angle between quanternions
81  T angle = std::acos(d);
82 
83  if(angle == 0) {
84  // If the angle is 0, then the output quaternion is q0
85  CopyVector4(q, q0);
86  } else {
87  T a = std::sin(angle*(1-t));
88  T b = std::sin(angle*t);
89  AddScaledVectors4(q, q0, qq1, a, b);
90  NormalizeQuaternion(q, q);
91  }
92  }

152  template<typename T>
153  inline void RotationMatrixToQuaternion(T q[4], const T r[9]) {
154  T qTemp[4];
155  double denom[4];
156 
157  denom[0] = r[0] - r[4] - r[8];
158  denom[1] = - r[0] + r[4] - r[8];
159  denom[2] = - r[0] - r[4] + r[8];
160  denom[3] = r[0] + r[4] + r[8];
161 
162  // Note: This differs slightly from typical implementations, since we normalize at the end rather than inline.
163  // This ensures greater numeric stability, and more precise normalization.
164  if ((denom[0] > denom[1]) && (denom[0] > denom[2]) && (denom[0] > denom[3])) {
165  qTemp[0] = 1 + denom[0];
166  qTemp[1] = r[1]+r[3];
167  qTemp[2] = r[2]+r[6];
168  qTemp[3] = r[7]-r[5];
169  }
170  else if ((denom[1] > denom[2]) && (denom[1] > denom[3])) {
171  qTemp[0] = r[1]+r[3];
172  qTemp[1] = 1 + denom[1];
173  qTemp[2] = r[5]+r[7];
174  qTemp[3] = r[2]-r[6];
175  }
176  else if ((denom[2] > denom[3])) {
177  qTemp[0] = r[2]+r[6];
178  qTemp[1] = r[5]+r[7];
179  qTemp[2] = 1 + denom[2];
180  qTemp[3] = r[3]-r[1];
181  }
182  else {
183  qTemp[0] = r[7]-r[5];
184  qTemp[1] = r[2]-r[6];
185  qTemp[2] = r[3]-r[1];
186  qTemp[3] = 1 + denom[3];
187  }
188 
189  NormalizeQuaternion(q, qTemp);
190  }

200  inline void RodriguesRotation(T R[9], const T axis[3], const T angle) {
201  const T& x = axis[0];
202  const T& y = axis[1];
203  const T& z = axis[2];
204 
205  T c = std::cos(angle);
206  T s = std::sin(angle);
207  T C = 1-c;
208 
209  T x2 = x*x;
210  T y2 = y*y;
211  T z2 = z*z;
212  T xy = x*y;
213  T xz = x*z;
214  T yz = y*z;
215 
216  R[0] = x2*C+c; R[1] = xy*C-z*s; R[2] = xz*C+y*s;
217  R[3] = xy*C+z*s; R[4] = y2*C+c; R[5] = yz*C-x*s;
218  R[6] = xz*C-y*s; R[7] = yz*C+x*s; R[8] = z2*C+c;
219  }

389  inline void ExpQuaternionFromSo3(T q[4], const T w[3]) {
390  T theta = sqrt(DotProduct3(w, w));
391  if( theta == (T) 0 ) {
392  IdentityQuaternion(q);
393  }
394  else if(theta <= DEFAULT_SAFE_SMALL_THETA_THRESHOLD) {
395  // Taylor expansion of sin(theta/2.0)/theta. next term +theta*theta*theta*theta/3840; //(2^5*5!)

397  ScaleVector3(q, w, ratio);
398  ImgToFullQuaternion(q, q);
399  }
400  else {
401  ScaleVector3<T>(q, w, sin(theta/2.0)/theta);
402  q[3] = cos(theta/2);
403  if( q[3]<0 ) {
404  ScaleVector4(q, q, (T) -1);
